带图例的饼图

            
var width = "100%";
var height = 300;
var color = d3.scaleOrdinal(d3.schemeCategory10);

var data = [
    {"city": "shanghai", "population": 10},
    {"city": "beijing", "population": 30},
    {"city": "tianjin", "population": 20},
    {"city": "shenzhen", "population": 45},
    {"city": "guangzhou", "population": 130}
];

//创建一个饼生成器
var pie = d3.pie()
        .value(function (d) {
            return d.population
        })(data);

//数据转换
var pieData = pie;
//弧生成器，内圈和外圈
var outerRadius = 100;
var innerRadius = 80;

//画曲线弧
var arc = d3.arc()
        .innerRadius(innerRadius)
        .outerRadius(outerRadius);

var svg = d3.select("#d3")
        .append("svg")
        .attr("width", width)
        .attr("height", height);

//g用于把相关元素进行组合的容器元素
var svgMain = svg.selectAll("g")
        .data(pieData)
        .enter()
        .append("g")
        .attr("transform", "translate(300,150)");

svgMain.append("path")
        .attr("fill", function (d, i) {
            return color(i)
        })
        .attr("d", function (d) {
            //调用弧生成器，得到路径值
            return arc(d)
        })
    //设置外扩过渡效果
        .on("mouseover", function (d, i) {
            d3.select(this)
                    .transition()
                    .duration(1000)
                    .ease(d3.easeBounce)
                    .attr("transform", function (d) {
                        //centroid返回的是弧形的重心与弧心的相对位置
                        return "translate(" + arc.centroid(d)[0] / 8 + "," + arc.centroid(d)[1] / 8 + ")";
                    });
        })
        .on("mouseout", function (d, i) {
            d3.select(this)
                    .transition()
                    .duration(1000)
                    .attr("transform", function (d) {
                        return "translate(" + [0, 0] + ")";
                    });
        });

//绘制图例区域
var legendArea = svg.append("g")
        .attr("transform", "translate(0,280)");

//绑定数据，设置每个图例的位置
var legend = legendArea.selectAll("g")
        .data(pieData)
        .enter()
        .append("g")
        .attr("transform", function (d, i) {
            return "translate(" + i * 120 + ",0)";
        });
//添加图例的矩形色块
legend.append("rect")
        .attr("width", 20)
        .attr("height", 20)
        .style("fill", function (d, i) {
            return color(i)
        });

//添加图例文字
legend.append("text")
        .attr("x", 24)
        .attr("y", 9)
        .style("fill", function (d, i) {
            return color(i)
        })
        .attr("dy", ".35em")
        .text(function (d, i) {
            return d.data.city + "[" + d.data.population + "]";
        });
